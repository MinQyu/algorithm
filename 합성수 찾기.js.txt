function solution(n) {
    let answer = 0;
    for(i=4; i<=n; i++){
        const tmp = [];
        for(j=1; j<=Math.sqrt(i); j++){
            if(i%j === 0){
                tmp.push(j);
                i/j !== j ? tmp.push(i/j) : null;
            }
            if(tmp.length>=3) break;
        }
        tmp.length>=3 ? answer+=1 : answer+=0;
    }
    return answer;
}
